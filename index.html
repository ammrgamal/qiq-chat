export default async function handler(req, res) { if (req.method !== "POST") { res.setHeader("Allow", ["POST"]); return res.status(405).json({ error: "Method not allowed" }); } try { const body = typeof req.body === "string" ? JSON.parse(req.body) : (req.body || {}); const messages = Array.isArray(body?.messages) ? body.messages : []; const upstream = await fetch("https://api.openai.com/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`, "Content-Type": "application/json", }, body: JSON.stringify({ model: "gpt-4o-mini", messages, temperature: 0.3, }), }); const raw = await upstream.text(); // لو OpenAI رجّع خطأ، رجّعه كـ JSON بدل ما نرجع Response فاضي if (!upstream.ok) { return res.status(upstream.status).json({ error: "upstream", raw }); } // تأكد إننا بنرجّع JSON صالح دايمًا const data = JSON.parse(raw); return res.status(200).json(data); } catch (err) { return res.status(500).json({ error: err.message }); } }
